date time format

moment().format('MMMM Do YYYY, h:mm:ss a'); // 十月 27日 2017, 12:36:41 下午
moment().format('dddd');                    // 星期五
moment().format("MMM Do YY");             // 10月 27日 17
moment().format('YYYY [escaped] YYYY');     // 2017 escaped 2017
moment().format();                        // 2017-10-27T12:36:41+08:00
